``` cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        unordered_map<int,int> freq;
        for(int i : nums){
            freq[i]++;
        }
        int res=-1;
        for(auto[num,i]:freq){
            if(i==1){
                res=num
                break
            }
        }
        return res;
    }   
};
```

:::
### 分析
- 時間複雜度：$\mathcal{O}(2n)$  
- 空間複雜度：$\mathcal{O}(n)$
