# 137.  
## 解答
1. hash map   
2. each bit mod 3  
3. bit xor

``` cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        unordered_map<int,int> freq;
        for(int i : nums){
            freq[i]++;
        }
        int res=-1;
        for(auto[num,i]:freq){
            if(i==1){
                res=num
                break
            }
        }
        return res;
    }   
};
```
``` cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int res=0;
        //int has 32 bit
        for(int i=0;i<32;i++){
            int total=0;
            // each nums=[a,a,a,b....]
            for(int j : nums){
                // sum all i-th bit
                total +=(j>>i & 1);
            }
            // if total % 3 ==1 then answer
            if(total %3){
                res |= (j<<i) ;
            }
        }
        return res;
    }   
};
```
