class Solution {
public:
    int numSquares(int n) {
        if(n==1) return 1;
        queue <int> q;
        q.push(n);
        int ans=0;
        while(!q.empty()){
            ans++;
            int sz=q.size();
            for(int i=0 ; i<sz ;i++){
                int remain = q.front();
                q.pop();
                int SqrtRemain= (int) sqrt(remain);
                if(SqrtRemain*SqrtRemain==remain) return ans;
                for(int j=SqrtRemain; j>0;j--){
                    q.push(remain-j*j);
                }
            }

        }
        throw;
    }
};
